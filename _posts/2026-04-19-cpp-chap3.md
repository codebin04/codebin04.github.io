		return;
	}

	data[next] = n;
	next++;

	for (int i = 0; i < next; i++) {
		cout << data[i] << ' ';
	}
}

double Storage::getAvg() {
	double sum = 0;
	for (int i = 0; i < next; i++) {
		sum += data[i];
	} 
	return sum/next;
}

	int getValue() { return value; }
	void loop();
};

void Calculator::calc(string op, int number) {

	if (op == "+") {
		value += number;
	}
	else if (op == "-") {
		value -= number;
	}
	else if (op == "*") {
		value *= number;
	}
	else if (op == "/") {
		if (number == 0) {
			cout << "0으로 나눌수없음" << endl;
			return;
		}
		value /= number;
	}	
}

	string op;

	}

class Pipe {
	int data[10];
	int index;

		for (int i = 0; i < 10; i++) data[i] = 0;
		index = 0;
	}
	int arrive(int n);
	void show();
	int shift();
};

int Pipe::arrive(int n) {
	int remove = 0;
	if (index == 10) { //파이프가 꽉 찼다면,
		remove = data[0];
		for (int i = 0; i <9; i++) { //저장된 정수들을 앞(왼쪽)이동
			data[i] = data[i + 1];
		}
		data[9] = n;//파이프 맨 뒤에 n 삽입
		return remove; //data[0]값을 리턴
	}
	else {//파이프가 꽉 안찼다면
		data[index] = n;//파이프의 맨 뒤에 n삽입
		index++;//인덱스 위치 이동
		return -1; // -1 리턴
	}
}

	for (int i = 0; i < 10; i++) {
		cout << data[i] << ' ';
	}

int Pipe::shift() {
	int front = data[0];
	for (int i = 0; i < 9; i++) { //저장된 정수들을 앞(왼쪽)이동
		data[i] = data[i + 1];
	}
	data[9] = 0;
	return front;
}
