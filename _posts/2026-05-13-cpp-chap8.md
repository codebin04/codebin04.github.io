```cpp
**Schedule.h**

#ifndef SCHEDULE_H
#define SCHEDULE_H

class Schedule{ //FCFS
protected:
	int count; //jobs 배열크기
	int* jobs; //방문한 아파트 동 번호
	int curLoc; //현재 택배기사 동 번호
public:
	Schedule(int curloc, int n[], int count);
	virtual ~Schedule();
	virtual int run(); //도착순서대로 택배처리
	int getJobs() { return count; }
	int getCurLoc() { return curLoc; }
};

#endif
```
```cpp
**Schedule.cpp**

#include <iostream>
#include <cstdlib>

#include "Schedule.h"
using namespace std;

Schedule::Schedule(int curloc, int n[], int count) {
	this->curLoc = curloc;

	this->jobs = new int[30]; // 0~29 동
	for (int i = 0; i < count; i++)
		jobs[i] = n[i];

	this->count = count;
}

Schedule::~Schedule() {
	delete[] jobs;
}

int Schedule::run() {
	int distance = 0;;

	cout << "현재 위치는 " << getCurLoc() << "동: 배달 시작 ...";
	for (int i = 0; i < count; i++) {
		cout << getCurLoc() << " -> ";
		curLoc = getCurLoc(); //현재위치파악
		distance += abs(curLoc - jobs[i]); //거리 전부 더하기
		curLoc = jobs[i]; //다음 동으로 이동
	}
	cout << getCurLoc() << endl;

	return distance;
}
```

```cpp
**ShortestFirst.cpp**

#include <iostream>
#include <cstdlib>
#include "ShortestFirst.h"
using namespace std;

ShortestFirst::ShortestFirst(int curloc, int n[], int count) :
	Schedule(curloc, n, count){}

int ShortestFirst::run() {
	int disSum = 0;
	
	cout << "현재 위치는 " << curLoc << "동: 배달 시작 ...";

	for (int i = 0; i < count; i++) {
		int minimum = 30; //culLoc 30에서 1빼도 30을 넘지못함
		int minIndex = 0; //반복문마다 초기화
		
		for (int d = 0; d < count; d++) {
			if (jobs[d] == -1) continue; // 이미 처리된 작업 스킵
			int dist = abs(curLoc - jobs[d]);//거리의 절댓값
			if (dist < minimum) { //최소거리가 같아도 작은인덱스먼저 방문 (ex: 1동-3동(내위치)-5동 ..1동먼저)
				minimum = dist; //최소거리
				minIndex = d; // 가장 가까운 인덱스 저장
			}
		}
		cout << getCurLoc() << " -> ";

		curLoc = jobs[minIndex]; //현재 위치를 최소거리 인덱스로 변경
		jobs[minIndex] = -1;// 처리 완료 표시
		disSum += minimum; //최소거리 더하기

	}
	cout << getCurLoc() << endl;

	return disSum;
}
```
