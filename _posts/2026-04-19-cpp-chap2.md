## 6번

```cpp
#include <iostream>
#include <cstring>
using namespace std;

void printDay(int startIdx) {
    // 시작 요일만큼 공백 출력
    for (int i = 0; i < startIdx; i++) {
        cout << "\t";
    }

    for (int i = 1; i <= 31; i++) {
        cout << i << "\t";

        // 줄바꿈 조건 (시작 요일 고려)
        if ((i + startIdx) % 7 == 0) {
            cout << endl;
        }
    }
}

int main() {
    string day;
    string week[7] = { "일","월","화","수","목","금","토" };

    cout << "3월 1일은 무슨 요일입니까 >> ";
    cin >> day;

    // 요일 출력
    for (int i = 0; i < 7; i++) {
        cout << week[i] << "\t";
    }
    cout << endl;

    // 입력한 요일의 인덱스 찾기
    int startIdx = -1;
    for (int i = 0; i < 7; i++) {
        if (week[i] == day) {
            startIdx = i;
            break;
        }
    }

    // 예외 처리 (잘못된 입력)
    if (startIdx == -1) {
        cout << "잘못된 요일 입력입니다." << endl;
        return 0;
    }

    printDay(startIdx);
}

```

#include <iostream>
#include <cstring>
using namespace std;

int main() {
	char buf[100];
	while (true) {
		int count = 0;

		cout << "영문 텍스트를 입력하세요 (빈칸 포함 가능)>>";
		cin.getline(buf, 100, '\n');

		if (strcmp(buf, "exit") == 0) {
			break;
		}
		
		for (int i = 0; buf[i]!='\0'; i++) {
			if (buf[i] != ' ' && (i == 0 || buf[i - 1] == ' ')) { //지금 글자가 공백이고, 첫글자거나 또는 이전 글자가 공백이라면
				count++;
			}
		}

		//단어의 길이 = 문자열의 길이 / 
		
		cout << "단어의 개수는 " << count << "개이다." << endl;
		
	}
}

	srand((unsigned)time(0));
	cout << "*** 구구단 맞추기 퀴즈입니다. ***" << endl;
	int count=0;

	while (true) {
		int input;

		int n = rand()%RAND_MAX;

		cout << n << 'x' << m << '=';
		cin >> input;
		if (input == n * m) {
			cout << "정답입니다 잘했습니다" << endl;

			count++;
			cout << count << "번 틀렸습니다.";

				break;

#include <iostream>
#include <string>
using namespace std;

int main() {
	string hwang = "황수희";
	string jang = "장수희";
	int hNum=0;
	int jNum=0;
	int count;

	string text;

	cout << "과대표 출마자 :" << hwang << ", " << jang << endl;
	
	while (true) {
		cout << "이름과 투표수>>";
		cin >> text;

		if (text == "그만") {
			cout << "최종 집계: " << hwang << ' ' << hNum << ", " << jang << ' ' << jNum << endl;
			if (hNum > jNum) {
				cout << "과 대표로 " << hwang << "가 선출되었습니다.";
				break;
			}
			else if (jNum > hNum) {
				cout << "과 대표로 " << jang << "가 선출되었습니다.";
				break;
			}
			else {
				cout << "투표 수가 동일합니다. 재투표하십시오";
				break;
			}
		}

		cin >> count;
		
		if (text == "황수희") {
			hNum += count;
		}
		else if (text == "장수희") {
			jNum += count;
		}
		else {
			cout << text << "은/는 출마자가 아닙니다." << endl;
			continue;
		}

	}

	cout << espresso << ' ' << esPrice << "원," << americano << ' ' << amePrice << "원," << cappucino << ' ' << cappuPrice << "원입니다." << endl;
	while (true) {
		if (sum >= 30000) {
			cout << "오늘 " << sum << "원을 판매하여 카페를 닫습니다. 내일봐요";
			break;
		}

		cout << "주문, 개수>>";
		cin >> menu >> ea;

		if (menu == "에스프레소") {
			sum += esPrice * ea;
			cout << esPrice * ea << "원입니다. 맛있게 드세요" << endl;
		}
		else if (menu == "아메리카노") {
			sum += amePrice * ea;

		}
		else if (menu == "카푸치노") {
			sum += cappuPrice * ea;
			cout << cappuPrice * ea << "원입니다. 맛있게 드세요" << endl;
		}
		else {
			cout << menu << "은/는 없습니다 다시 입력하세요." << endl;

		}
