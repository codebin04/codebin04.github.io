## 1번
```cpp
#include <iostream>
using namespace std;

#define SIZE 5
int main() {
	double* arr = new double[SIZE];

	cout << "온도 5개 입력>>";
	for (int i = 0; i < SIZE; i++) {
		cin >> arr[i];
	}
	double sum = 0;
	for (int i = 0; i < SIZE; i++) {
		sum += arr[i];
	}
	cout << "평균은 " << sum / SIZE;
	
	delete[] arr;
}
```

## 2번
```cpp
#include <iostream>
using namespace std;

int main() {
	int n;
	cout << "구입할 물품의 개수 >> ";
	cin >> n;

	int* p = new int[n];

	cout << "물품 " << n << "개의 가격 입력>>";
	for (int i = 0; i < n; i++) {
		cin >> p[i];
	}
	int max = p[0];
	int min = p[0];

	for (int i = 0; i < n; i++) {
		if (p[i] > max) {
			max = p[i];
		}
	}

	for (int i = 0; i < n; i++) {
		if (p[i] < min) {
			min = p[i];
		}
	}

	cout << "제일 싼 가격은" << min << endl;
	cout << "제일 비싼 가격은" << max << endl;
	delete[] p;
}
```

## 3번
```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
	string line;
	string dest ="";
	cout << "텍스트 입력 >>";
	getline(cin, line, '\n');

	int index = line.size()-1;
	cout << index << endl;
	int i = 0;

	while (true) {
		if (i > index) { // i가 인덱스 끝까지 도달하면 종료
			break;
		}

		if (line[i] == ' ') {// 현재 문자가 공백이면
			dest.append(1, ' ');// 공백 1개만 추가
			while (i <= index && line[i] == ' ') { // 연속 공백 건너뜀
				i++;
			}
		}
		else {// 공백이 아니면
			dest.append(1, line[i]);// 그대로 추가
			i++;
		}
	}
	cout << dest;
}
```

## 5번
```cpp
#include <iostream>
using namespace std;

class Container {
	int* p;
	int size;
public:
	Container(int size);
	~Container();
	void read();
	void write();
	void rotate();
	double avg();
};

Container::Container(int size) {
	this->size = size;
	p = new int[size];
}

Container::~Container() {
	delete[] p;
}

void Container::read() {
	cout << "정수 " << size << "개 입력>>";
	for (int i = 0; i < size; i++) {
		cin >> p[i];
	}
}

void Container::write() {
	for (int i = 0; i < size; i++) {
		cout << p[i] << ' ';
	}
	cout << endl;
}

void Container::rotate() {
	int temp = p[size - 1]; //끝 숫자 저장

	for (int i = size-1; i > 0; i--) { //역반복문 높은수 -> 낮은수
		p[i] = p[i-1]; //모든 배열을 오른쪽으로 회전
	}

	p[0] = temp; //처음 값에 끝 숫자 저장하기
}

double Container::avg() {
	double sum = 0;
	for (int i = 0; i < size; i++) {
		sum += p[i];
	}

	return sum / size;
}
int main() {
	Container c(10);
	c.read();
	c.write();
	c.rotate();
	c.write();
	cout << "평균 >> " << c.avg() << endl;
}
```
