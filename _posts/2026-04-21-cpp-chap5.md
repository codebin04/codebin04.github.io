		min = b;

		return false;
	}

#include <iostream>

class Contest {
	string* teams;
	int* scores;
	int size;
public:
	Contest(int size);
	Contest(const Contest& src);
	~Contest();
	void read();
	int getSize() { return size; }

Contest::Contest(int size) {
	this->size = size;
	teams = new string[size];
	scores = new int[size];
}

Contest::Contest(const Contest& src) {
	this->size = src.size;
	teams = new string[size];
	scores = new int[size];

	if (!teams || !scores) {
		exit(1);
	}
	for (int i = 0; i < size; i++) {
		teams[i] = src.teams[i];
		scores[i] = src.scores[i];
	}
}

Contest::~Contest() {
	if (teams) delete[] teams;
	if (scores) delete[] scores;
}

void Contest::read() {
	cout << size << "개 입력(팀명 점수) >> ";
	for (int i = 0; i < size; i++) {
		cin >> teams[i] >> scores[i];
	}
}

string decideGoldAward(Contest c) {
	int index;
	int maximum = 0;
	for (int i = 0; i < c.getSize(); i++) {
		if (c.getScore(i) > maximum) {
			maximum = c.getScore(i);
			index = i;
		}
	}
	return c.getTeam(index);
}

int main() {
	Contest codingTest(6);
	codingTest.read();
	string team = decideGoldAward(codingTest);
	cout << "금상은 " << team << "팀입니다." << endl;
}
