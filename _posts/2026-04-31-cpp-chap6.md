#include <iostream>
#include <string>
using namespace std;

bool equals(string x, string y, int n = 0, string z = "") {
	if (n > 0 && z != "") { //몇번째 인덱스인지 값이 존재하고, z가 존재할때
		return x.substr(0, n) == y.substr(0, n) && x.substr(0, n) == z.substr(0, n);
	}
	else if (n > 0) { //인덱스값만 존재할때
		return x.substr(0, n) == y.substr(0, n);
	}
	else { //x와 y만 주어질때
		return x == y;
	}
}

int main() {
	string x = "Prof. Hwang";
	string y = "Prof. Kim";
	string z = "Prof. Lee";

	if (equals(x, y)) cout << x << " 와 " << y <<" 는 같음" << endl;
	else {
		cout << "다름" << endl;
	}
	if (equals(x, y, 3)) cout << "앞 3글자 같음" << endl;
	if (equals(x, y, 5, z)) cout << x << ", " << y << ", " <<z << " 의 앞 5글자 같음" << endl;
}

#include <iostream>
#include <string>
using namespace std;

string erase(string text, string target =" ",string replace = "") { //text문장에 존재하는 erase를 모두 replace로 대체
	int i = 0;
	while (true) {
		int index = text.find(target); //빈칸의 인덱스값

		if (index == string::npos) {
			break;
		}
		text.replace(index, target.length(), replace); //index값에 있는 target을 replace로 대체, 존재하는 모든것
	}
	return text;
}

int main() {
	string a = erase("Hello   world,   Yaho"); //모든 " "(띄어쓰기) 지우기

#include <iostream>
using namespace std;
#include "Vector.h"

Vector::Vector(int size, int value) {
	this->size = size;
	mem = new int[size];
	if (mem == NULL) { //없으면 감점요소
		cout << "동적할당 실패. 더이상 실행 할 수 없음";
		exit(0); //종료
	}
	for (int i = 0; i < size; i++) {
		mem[i] = value;
	}
}

Vector::~Vector() { //동적배열 후 소멸자 작성 안할시 감점
	delete[] mem;
}

void Vector::show(int n) {
	if (n == 0) { //매개변수가 없으면
		n = size;
	}
	for (int i = 0; i < n; i++) {
		cout << mem[i] << ' ';
	}
	cout << endl;
}

void GameBoard::increase(int point, const string& team) {
	if (team == "청군") {
		this ->blue += point;
	}
	else if (team == "백군") {
		this ->white += point;
	}
	else if (team == "입력없음") {
		this ->blue += point;
		this ->white += point;
	}
	else {
		cout << "잘못된 입력";
		exit(1);
	}
}

void GameBoard::show(const string& team) {
	if (team == "입력없음") {
		cout << "청군 :" << blue << ", 백군 :" << white << endl;
	}
	else if (team == "청군") {
		cout << "청군 :" << blue << endl;
	}
	else if (team == "백군") {
		cout << "백군 :" << white << endl;
	}
}

#include <iostream>
#include <cstdlib>
using namespace std;

class Random {
	Random() {}
public:
	static void seed() { srand((unsigned)time(0)); } //시드 설정
	static int nextInt(int min = 0, int max = 32767);

	static char nextAlphabet();
	static double nextDouble();
};

int Random::nextInt(int min, int max) {
	int range = max - min;
	if (range < 0)
		return -1;
	int n = rand(); // 0에서 RAND_MAX(32767) 사이의 랜덤한 정수 발생
	n = (n % (range + 1)) + min;
	return n;
}
char Random::nextAlphabet() {
	int n = nextInt(0, 25); // 알파벳 26글자
	int s = nextInt() % 2; // 대소문자
	if (s == 0) // 대문자
		return 'A' + n;
	else // 소문자
		return 'a' + n;
}
double Random::nextDouble() {
	return ((double)rand()) / RAND_MAX;
}

int main() {
	Random::seed();

	cout << "1에서 100까지 랜덤한 정수 10개를 출력합니다" << endl;
	cout << "->";
	for (int i = 0; i < 10; i++) {
		cout << Random::nextInt(1,100) << ' ';
	}
	cout << endl;

	cout << "알파벳을 랜덤하게 10개 출력합니다" << endl;
	cout << "->";
	for (int i = 0; i < 10; i++) {
		cout << Random::nextAlphabet() << ' ';

	cout << "랜덤한 실수 10개를 출력합니다" << endl;

#include <iostream>
#include <string>
using namespace std;


class WordUtility {
public:
	static int wordCount(string s);
    static string getWord(string s, int index);
};

int WordUtility::wordCount(string s) {
    int count = 0;
    bool inWord = false;
    //앞이나 뒤가 공백이면, 단어임
    for (int i = 0; i<s.length(); i++) { //공백이 아니면 단어임.
        if (s[i] != ' ' && !inWord) {
            count++;// 새 단어 시작
            inWord = true;
        }
        else if (s[i] == ' ') { //공백이면, 단어가 아님
            inWord = false;
        }
    }
    return count;
}

string WordUtility::getWord(string s, int index) {
    int count = 0;
    string word = "";

    for (int i = 0; i <= s.length(); i++) { //문자열 첨부터 끝까지 훑기

        if (s[i] != ' ' && i != s.length()) { //문자라면.끝이아니라면.
            word += s[i];// 문자 저장
        }
        else {// 공백이거나 문자열 끝이면
            if (word != "") {
                count++; //다음 단어 인덱스로 카운트 업
                if (count == index) //만약 찾는 순서와 같다면
                    return word;  // 원하는 번째 반환
                word = "";  // 단어 초기화
            }
        }
    }
    return "";
}

int main() {
	int n = WordUtility::wordCount("I love C++");
	cout << "단어 개수는 " << n<< endl;

    string word = WordUtility::getWord("I love C++", 3); //3번째 단어 C++리턴
    if (word == "") {
        cout << "3번째 단어는 없습니다." << endl;
    }
    else {
        cout << "3번째 단어는 " << word << endl;
    }
}

#include <iostream>
#include <string>
using namespace std;

class Trace {
	static string traces[1000];  // 모든 호출이 공유하는 공간
	static int count;
public:
	static void put(string tag, string debug);
	static void print(string tag= "");
};

string Trace::traces[1000];
int Trace::count = 0;

void Trace::put(string tag, string debug) {
	traces[count] = tag + " : " + debug;
	count++;
}

void Trace::print(string tag) {
	int i = 0;
	while (traces[i] != "") { //배열의 끝까지 반복
		if (tag == "" || traces[i].find(tag) == 0) { //태그가 없음 or 태그가 tag로 시작하면
			//ex : if tag == "main()"
			//ex : traces[0] = "main()",,, -> true -> tarce[0]출력
			//ex : traces[1] = "sumTo()",,, -> false -> 미출력
			cout << traces[i] << endl; //태그가 없으면, 전부 출력 / tag가 일치하면, 해당하는 i번째 인덱스들만 출력 or조건문에 의해서
		}
		i++; //다음 인덱스로 이동
	}
}

int sumTo(int start, int end) {
	Trace::put("sumTo()", "start=" + to_string(start));
	Trace::put("sumTo()", "start=" + to_string(end));
	int sum = 0;
	for (int i = start; i <= end; i++) {
		sum += i;
		Trace::put("sumTo()", "sum=" + to_string(sum));
	}
	return sum;
}

int main() {
	Trace::put("main()", "프로그램 시작"); // trace에 디버깅정보 저장
	int n = sumTo(1, 5);
	Trace::put("main()", "n=" + to_string(n));
	Trace::print("main()");
	
	cout << endl;
	Trace::print();
}
