#include <iostream>
using namespace std;

class Chart {
	int* p = NULL;
	int size;
	int lastIndex = 0; //인덱스 마지막위치
public:
	Chart(int size = 0);
	~Chart();
	void show();

	friend bool operator ! (const Chart& op1);
	friend Chart& operator >> (Chart& op1, int data);
	friend int operator ~ (const Chart& op1);
};

Chart::Chart(int size) {
	this->size = size;
	p = new int[size];
	if (p == NULL) {
		cout << "동적할당오류" << endl;
		exit(0);
	}
}

Chart::~Chart() {
	if (p != NULL) { //오류방지
		delete[] p;
	}
}

void Chart::show() {
	if (lastIndex == 0) {
		cout << "보여줄 차트가 비었습니다." << endl;
		return;
	}

	cout << "현재 차트 : ";
	for (int i = 0; i < lastIndex; i++) {
		cout << p[i] << ' ';
	}
	cout << endl;
}

bool operator !(const Chart& op1) {
	if (op1.lastIndex == 0) return true;
	else return false;
}

Chart& operator >> (Chart& op1, int data) {
	op1.p[op1.lastIndex] = data;
	op1.lastIndex++;
	return op1;
}

int operator ~ (const Chart& op1) {
	int sum = 0;

	if (op1.lastIndex == 0) return 0;

	for (int i = 0; i < op1.lastIndex; i++) {
		sum += op1.p[i];
	}
	return sum / op1.lastIndex;
}

int main() {
	Chart chart(100);
	if (!chart) cout << "차트가 비었습니다." << endl;

	chart.show();

	int n;
	cout << "정수 4개 입력>>";
	for (int i = 0; i < 4; i++) {
		cin >> n; // 정수 입력
		chart >> n;
	}
	int avg = ~chart;
	cout << "평균은 " << avg << endl;
	chart.show();
}
